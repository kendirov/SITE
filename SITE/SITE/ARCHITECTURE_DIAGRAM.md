## Key Algorithms

### 1. Vacuum Strategy (Pagination)
```javascript
async function getStockAlgoStats(date, onProgress) {
  const allRecords = []
  let currentStart = 0
  let hasMoreData = true

  while (hasMoreData) {
    // Fetch batch
    const batch = await fetchMoex(endpoint, {
      date,
      start: currentStart,
      limit: 100
    })

    // Parse batch
    const records = transformResponse(batch)

    // Check if done
    if (records.length === 0 || records.length < 100) {
      hasMoreData = false
    }

    // Aggregate
    allRecords.push(...records)

    // Report progress
    if (onProgress) {
      onProgress(allRecords.length, allRecords.length + 100)
    }

    // Next batch
    currentStart += 100

    // Throttle
    await sleep(100)
  }

  return allRecords
}
```
